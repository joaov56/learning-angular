    let app= angular.module('mainApp', ['ngRoute']);

    app.config(function($routeProvider){
        $routeProvider
        .when('/', {
            templateUrl: 'Welcome user!'
        })
        .when('/anotherPage', {
            templateUrl: 'Welcome, again'
        })
        .otherwise({
            redirectTo: '/'
        })
    })

        }).then(function(response){
            $scope.persons= response.data.records
        }).catch(function(err){
            console.log(err);
        })

    })
